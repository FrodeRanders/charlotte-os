\chapter{Current Status}

\section{Verified (boots on QEMU, zero panics)}

\begin{itemize}
    \item \textbf{AArch64:} boots, all self-tests pass, the async-syscall demo
        succeeds with \texttt{Ok(42)} via exit-observer, real-EL0 user thread
        executes at EL0 and dispatches syscalls. The sitas
        \texttt{basic\_kv} user binary runs through \texttt{crt0/cmain},
        spawns a shard via SVC \#7, and reports \texttt{total\_len == 3}.
    \item \textbf{x86-64:} boots with \texttt{-cpu max}, all self-tests pass,
        the async-syscall demo succeeds with \texttt{Ok(42)} via exit-observer.
        (x86-64 ring-3 SYSCALL userspace is implemented but not yet exercised
        with a user thread.)
    \item \textbf{Completion subsystem:} submit, complete, poll, wait, cancel,
        close, \texttt{submit\_worker} (exit-observer), CQ ring, CQ ring
        integrated with AS, non-lossy CQ overflow backlog.
    \item \textbf{Syscall dispatch:} AArch64 SVC handler for completion
        operations, thread exit, LP-pinned spawn, CQ wait, and mailbox
        capability operations. The caller's address space is taken from
        \texttt{TrapFrame.asid}, never from a userspace ASID parameter. x86-64
        SYSCALL trampoline exists, connected to the dispatch table.
    \item \textbf{IPI:} bounded per-LP queues, backpressure, typed closure
        dispatch, both architectures unstubbed. Cross-core SGI delivery works
        on QEMU AArch64 GICv3 with \texttt{-smp 2}; the boot gate exercises
        kernel \texttt{ShardMailbox} fan-out, EL0 cross-LP completion, and EL0
        ping-pong over mailbox endpoint capabilities on both the default
        accelerator path and explicit TCG.
    \item \textbf{ShardLocal<T> / ShardMailbox<M>:} implemented and tested
        in-kernel.
    \item \textbf{Scheduler:} RoundRobin, quantum timer, block/yield/wake,
        deferred reaper, per-LP thread pinning (\texttt{submit\_to\_lp}).
    \item \textbf{sitas integration:} sitas-core (no\_std executor) compiles
        for aarch64-unknown-none. sitas-charlotte (ReactorBackend +
        ShardRuntime::spawn\_shard via SVC \#7) links against the kernel's
        syscall ABI. The current user binary runs \texttt{basic\_kv}, creates a
        \texttt{ShardedKv}, spawns a shard thread, and verifies the result from
        EL0.
    \item \textbf{CI:} GitHub Actions builds both architectures with
        \texttt{-D warnings}, QEMU boot gate.
\end{itemize}

\section{Known limitations}

\begin{itemize}
    \item \textbf{x86-64 ring-3 userspace:} SYSCALL entry trampoline exists,
        GS base and IRETQ return path remain to be wired. The EL0 test and
        real-EL0 user thread are AArch64-only.
    \item \textbf{General userspace loader:} the sitas smoke now uses a
        segment-aware ELF loader for its embedded user image, including
        per-segment page permissions and zero-filled BSS. It is still a
        fixed-address self-test harness for config, CQ, input, and heap pages,
        not a general process loader with argv/env/auxv-style setup.
\end{itemize}

\section{Repositories and branches}

\begin{itemize}
    \item \textbf{sitas} (\texttt{reactor-handle-seam}):
        \texttt{ReactorBackend} trait, ABI reference model \\
        (\texttt{charlotte\_abi.rs}), \texttt{sitas-core} no\_std crate
        (ringbuf, ShardRuntime, ShardLocal, ShardMailbox, basic\_kv), 
        \texttt{sitas-charlotte} crate (ReactorBackend + ShardRuntime backend,
        with ASID kept out of the userspace API).
    \item \textbf{CharlotteOS} (\texttt{dev}): consolidated kernel ---
        completion subsystem, syscall dispatch, bounded IPI, ShardLocal,
        ShardMailbox, CQ ring, real-EL0 test, exit-observer completion,
        SPAWN\_THREAD syscall, sitas user binary, developer manual (31-page
        PDF).
\end{itemize}

\section{Build and boot}

\textbf{AArch64:}
\begin{verbatim}
cargo build --package catten \
  --target target_specs/aarch64-unknown-none-catten.json \
  --no-default-features --features acpi
./scripts/boot-smp1.sh
\end{verbatim}

\textbf{x86-64:}
\begin{verbatim}
cargo build --package catten \
  --target target_specs/x86_64-unknown-none-catten.json \
  --no-default-features --features acpi
./scripts/boot-x86.sh
\end{verbatim}

\textbf{sitas user binary (requires nightly + build-std):}
\begin{verbatim}
cargo +nightly build --manifest-path crates/catten-user/Cargo.toml \
  --target crates/catten-user/aarch64-unknown-none.json \
  -Zbuild-std=core,alloc,compiler_builtins
\end{verbatim}
